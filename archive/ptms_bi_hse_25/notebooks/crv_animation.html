def pdf(x):
    if (x < 1):
        return 0
    elif (x >= 1) and (x <= 5):
        return (5 / 4) * (1 / (x ** 2))
    else:
        return 0

def cdf(x):
    if (x < 1):
        return 0
    elif (x >= 1) and (x <= 5):
        return (5*x - 5) / (4 * x)
    else:
        return 1

l = 5.6+1.53333
def pdf(x):
    l = 5.6+1.53333
    if (x < 0):
        return 0
    elif (x>=0) and (x<4.4):
        return 0.12
    elif (x>=4.4) and (x<=5.6):
        return (-0.5 * (x**2) + 5*x - 12.2)
    elif (x>5.6) and (x<l):
        return 0.12
    else:
        return 0


def cdf(x):
    l = 5.6+1.53333
    if (x < 0):
        return 0
    elif (x>=0) and (x<=4.4):
        return 0.12 * x
    elif (x>4.4) and (x<=5.6):
        return (cdf(4.4) + -0.166667 * (x**3) + 2.5 * (x**2) - 12.2 * x + 19.4773)
    elif (x>5.6) and (x<l):
        return cdf(5.6) + (x-5.6) * 0.12
    else:
        return 1
