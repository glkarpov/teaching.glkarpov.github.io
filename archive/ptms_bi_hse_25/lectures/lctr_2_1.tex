   \textbf{Для 1 и 2 варианта: Рассмотрим две случайные величины $X$ и $Y$.}

   \textbf{Обе они могут принимать значения $0$, $1$ и $2$.}

   Восстановите таблицу совместного распределения случайных величин $X$ и $Y$.
    
   Найдите $\mathbb{E}(X), \mathbb{E}(Y), \operatorname{Var}(X), \operatorname{Cov}(X, Y)$.

    \textbf{1 вариант: Фамилии от А до Г:}

    \begingroup\Large
    $$
    \begin{aligned}
    \mathrm{P}(X=0) & =0.2 \quad \text { and } \mathrm{P}(X=1)=0.3, \\
    \mathrm{P}(Y=0) & =0.2 \quad \text { and } \mathrm{P}(Y=1)=0.3, \\
    \mathrm{P}(Y=0 \mid X=0) & =0.5 \quad \text { and } \mathrm{P}(Y=0 \mid X=1)=0.2, \\
    \mathrm{P}(Y=1 \mid X=0) & =0.4 \quad \text { and } \mathrm{P}(Y=1 \mid X=1)=0.3 .
    \end{aligned}
    $$
    \endgroup

    Найдите $P(Y = 1 \mid Y + X > 2)$.
    
    Постройте условное распределение случайной величины $X$ при условии $Y=0$

    \vspace{1.5cm}

    \textbf{2 вариант: Фамилии от Д до К:}

    $$
    \begin{aligned}

    \mathrm{P}(Y=0) & =0.25 \quad \text { and } \mathrm{P}(Y=1)=0.35, \\
    \mathrm{P}(Y=0 \mid X=0) & =0.2 \quad \text { and } \mathrm{P}(Y=0 \mid X=1)=0.25, \\
    \mathrm{P}(Y=1 \mid X=0) & =0.4 \quad \text { and } \mathrm{P}(Y=1 \mid X=1)=0.25 .
    \end{aligned}
    $$
    \endgroup
